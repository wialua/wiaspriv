-- Teleport Fonksiyonu
local function tp(...)
    local plr = game.Players.LocalPlayer
    local args = {...}
    if typeof(args[1]) == "number" and args[2] and args[3] then
        args = Vector3.new(args[1], args[2], args[3])
    elseif typeof(args[1]) == "Vector3" then
        args = args[1]
    elseif typeof(args[1]) == "CFrame" then
        args = args[1].Position
    end
    local dist = (plr.Character.HumanoidRootPart.Position - args).Magnitude
    game:GetService("TweenService"):Create(
        plr.Character.HumanoidRootPart,
        TweenInfo.new(dist / 1000, Enum.EasingStyle.Linear),
        { CFrame = CFrame.new(args) }
    ):Play()
end

-- üéØ Geli≈ümi≈ü ve Stabil Hitbox (Se√ßilebilir V√ºcut Par√ßasƒ±)
local HitboxSection = HitboxTab:CreateSection({ Name = "Ayarlar" })
local selectedTeams = {}
local isHitboxEnabled = false
local lastSize = 3.8
local lastTransparency = 0.5
local lastColor = Color3.fromRGB(255, 0, 0)
local selectedBodyPart = "Head" -- Varsayƒ±lan olarak ba≈ü
local hitboxConnection -- Baƒülantƒ±yƒ± saklayacaƒüƒ±mƒ±z deƒüi≈üken
local hitboxToggle -- Hitbox toggle butonunu saklayacaƒüƒ±z

-- Se√ßilebilecek v√ºcut par√ßalarƒ±
local bodyParts = {"Head", "Torso", "HumanoidRootPart", "UpperTorso", "LowerTorso"}

-- V√ºcut par√ßalarƒ±na hitbox uygulama fonksiyonu
local function applyHitbox(player)
    if player and player.Character then
        local part = player.Character:FindFirstChild(selectedBodyPart)
        
        if part then
            part.Size = Vector3.new(lastSize, lastSize, lastSize)
            part.Transparency = lastTransparency
            part.BrickColor = BrickColor.new(lastColor)
            part.CanCollide = false
            part.Material = Enum.Material.ForceField
        end
    end
end

-- V√ºcut par√ßalarƒ±na hitbox sƒ±fƒ±rlama fonksiyonu
local function resetHitbox(player)
    if player and player.Character then
        local part = player.Character:FindFirstChild(selectedBodyPart)
        
        if part then
            -- Varsayƒ±lan deƒüerler
            part.Size = Vector3.new(2, 2, 2)
            part.Transparency = 0
            part.BrickColor = BrickColor.new("Medium stone grey")
            part.CanCollide = true
            part.Material = Enum.Material.Plastic
        end
    end
end

-- T√ºm oyunculardaki hitbox'larƒ± g√ºncelleme fonksiyonu
local function updateHitboxes()
    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer and player.Team and table.find(selectedTeams, player.Team.Name) then
            -- Eski hitbox'ƒ± sƒ±fƒ±rlayƒ±p yeni hitbox'ƒ± uyguluyoruz
            resetHitbox(player)
            applyHitbox(player)
        else
            resetHitbox(player) -- Takƒ±m dƒ±≈üƒ± oyuncularda hitbox'ƒ± sƒ±fƒ±rlƒ±yoruz
        end
    end
end

-- Hitbox'ƒ± a√ßma/kapama toggle'ƒ±
hitboxToggle = HitboxSection:AddToggle({
    Name = "Hitbox Aktif",
    Value = false,
    Callback = function(state)
        isHitboxEnabled = state
        if state then
            -- Eƒüer hitbox aktifse, RenderStepped'e baƒülanalƒ±m
            if hitboxConnection then
                hitboxConnection:Disconnect()
            end
            hitboxConnection = RunService.RenderStepped:Connect(updateHitboxes)
        else
            -- Eƒüer hitbox kapalƒ±ysa, baƒülantƒ±yƒ± keselim
            if hitboxConnection then
                hitboxConnection:Disconnect()
                hitboxConnection = nil
            end
            -- Hitbox'ƒ± sƒ±fƒ±rlayalƒ±m
            for _, player in ipairs(Players:GetPlayers()) do
                resetHitbox(player)
            end
        end
    end
})

-- Boyut, ≈üeffaflƒ±k ve renk ayarlarƒ±
HitboxSection:AddSlider({ Name = "Boyut", Min = 1, Max = 5, Value = lastSize, Callback = function(v) lastSize = v end })
HitboxSection:AddSlider({ Name = "≈ûeffaflƒ±k", Min = 0, Max = 1, Value = lastTransparency, Callback = function(v) lastTransparency = v end })
HitboxSection:AddColorPicker({ Name = "Renk", Value = lastColor, Callback = function(v) lastColor = v end })

-- V√ºcut par√ßasƒ± se√ßimi dropdown'u
HitboxSection:AddDropdown({
    Name = "V√ºcut Par√ßasƒ± Se√ß",
    List = bodyParts,
    Callback = function(selectedPart)
        -- V√ºcut par√ßasƒ± deƒüi≈ütiƒüinde √∂nce hitbox'ƒ± kapatƒ±yoruz
        hitboxToggle:Set(false)  -- Hitbox toggle'ƒ±nƒ± kapatƒ±yoruz
        selectedBodyPart = selectedPart
        if isHitboxEnabled then
            -- Eƒüer hitbox aktifse, baƒülantƒ±yƒ± kesip yeni bir baƒülantƒ± kuruyoruz
            if hitboxConnection then
                hitboxConnection:Disconnect()
            end
            -- Hitbox'ƒ± yeniden ba≈ülatƒ±yoruz
            hitboxConnection = RunService.RenderStepped:Connect(updateHitboxes)
            -- Yeni par√ßa se√ßildikten sonra toggle'ƒ± yeniden aktif ediyoruz
            hitboxToggle:Set(true)  -- Toggle'ƒ± tekrar a√ßƒ±yoruz
        end
    end
})

-- Takƒ±m se√ßimi toggle'larƒ±
local TeamSection = HitboxTab:CreateSection({ Name = "Takƒ±m Se√ßimi" })
for _, team in ipairs({"Prisoners", "Police", "Criminals", "Heroes", "Villains"}) do
    TeamSection:AddToggle({
        Name = team,
        Value = false,
        Callback = function(value)
            if value then
                -- Takƒ±m se√ßildiƒüinde, bu takƒ±mƒ± ekliyoruz
                if not table.find(selectedTeams, team) then
                    table.insert(selectedTeams, team)
                end
            else
                -- Takƒ±m kaldƒ±rƒ±ldƒ±ƒüƒ±nda, bu takƒ±mƒ± listeden √ßƒ±karƒ±yoruz
                for i, t in ipairs(selectedTeams) do
                    if t == team then
                        table.remove(selectedTeams, i)
                        break
                    end
                end
            end

            -- Takƒ±m se√ßildiƒüinde hitbox'larƒ± g√ºncelliyoruz
            if isHitboxEnabled then
                if hitboxConnection then
                    hitboxConnection:Disconnect()
                end
                hitboxConnection = RunService.RenderStepped:Connect(updateHitboxes)
            end
        end
    })
end
