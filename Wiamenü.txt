local HitboxTab = PepsisWorld:CreateTab({
    Name = "Hitbox"

local HitboxSection = HitboxTab:CreateSection({
    Name = "Head Hitbox"
})

local isHitboxEnabled = false
local lastSize = 3.8
local lastTransparency = 0.5
local lastColor = Color3.fromRGB(255, 0, 0)
local HitboxConnection

local selectedTeams = {}
local teams = {"Prisoners", "Police", "Criminals", "Heroes", "Villains"}
local TeamCheckboxSection = HitboxTab:CreateSection({
    Name = "TakÄ±m SeÃ§imi"
})

for _, team in ipairs(teams) do
    TeamCheckboxSection:AddToggle({
        Name = team,
        Value = false,
        Callback = function(value)
            if value then
                table.insert(selectedTeams, team)
            else
                for i, t in ipairs(selectedTeams) do
                    if t == team then
                        table.remove(selectedTeams, i)
                        break
                    end
                end
            end
        end
    })
end

-- Functions for Hitbox
local function applyHitboxToAllPlayers()
    for _, player in ipairs(Players:GetPlayers()) do
        if player.Character and player.Character:FindFirstChild("Head") then
            local head = player.Character:FindFirstChild("Head")
            if player ~= Players.LocalPlayer then
                head.Size = Vector3.new(lastSize, lastSize, lastSize)
                head.Transparency = lastTransparency
                head.BrickColor = BrickColor.new(lastColor)
            end
        end
    end
end

local function resetHitboxForAllPlayers()
    for _, player in ipairs(Players:GetPlayers()) do
        if player.Character and player.Character:FindFirstChild("Head") then
            local head = player.Character:FindFirstChild("Head")
            if player ~= Players.LocalPlayer then
                head.Size = Vector3.new(2, 1, 1)
                head.Transparency = 0
                head.BrickColor = BrickColor.new("Bright red")
            end
        end
    end
end

local function expandHitboxForSelectedTeams()
    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= Players.LocalPlayer then
            if player.Character and player.Character:FindFirstChild("Head") then
                local head = player.Character:FindFirstChild("Head")
                for _, selectedTeam in ipairs(selectedTeams) do
                    if player.Team and player.Team.Name == selectedTeam then
                        head.Size = Vector3.new(lastSize, lastSize, lastSize)
                        head.Transparency = lastTransparency
                        head.BrickColor = BrickColor.new(lastColor)
                    end
                end
            end
        end
    end
end

local isAimbotLocked = false

local function aimAtPlayer(player)
    if player.Character and player.Character:FindFirstChild("UpperTorso") then
        local torso = player.Character.UpperTorso
        local targetPos = torso.Position
        game.Workspace.CurrentCamera.CFrame = CFrame.new(game.Workspace.CurrentCamera.CFrame.Position, targetPos)
    end
end

UserInputService.InputBegan:Connect(function(input)
    if input.KeyCode == Enum.KeyCode.E then
        if isAimbotActive then
            isAimbotLocked = not isAimbotLocked
            if isAimbotLocked then
                AimbotConnection = RunService.RenderStepped:Connect(function()
                    for _, player in pairs(Players:GetPlayers()) do
                        if player ~= Players.LocalPlayer then
                            if player.Team and table.find(selectedAimbotTeams, player.Team.Name) then
                                aimAtPlayer(player)
                            end
                        end
                    end
                end)
            else
                if AimbotConnection then
                    AimbotConnection:Disconnect()
                    AimbotConnection = nil
                end
            end
        end
    end
end)
